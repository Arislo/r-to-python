```r
#外部ライブラリのインストール
install.packages("ライブラリ名")
#例
install.packages("tidyverse")
#使用するとき
library("ライブラリ名")
#例
library("tidyverse")

#一行で書きたい
pacman::p_load("ライブラリ名")
#例
pacman::p_load(dplyr, tidyr, dbplyr, magrittr, ggplot2)
```
